#include <iostream>
using namespace std;
int editDist(string inp,string out){
    
   int m = inp.length();
   int n = out.length();
   int dp[100][100];

   dp[0][0]=0;
   for(int i=1;i<=n;i++) dp[0][i] = dp[0][i-1] + 1;
   for(int j=1;j<=m;j++) dp[j][0] = dp[j-1][0] + 1;

   for(int i=1;i<=m;i++){
       for(int j=1;j<=n;j++){
           int q1 = dp[i-1][j-1];  //replacement
           int q2 = dp[i-1][j];    //deletion
           int q3 = dp[i][j-1];    //insertion
           dp[i][j]=min(q1,min(q2,q3)) + (inp[i-1]!=out[j-1]);
       }
   }
   return dp[m][n];  
}

int main() {
    string str1,str2;
    cin>>str1>>str2;

    int ans = editDist(str1,str2);
    cout<<ans;
}
