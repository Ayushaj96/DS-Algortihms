	{
		while (n%i == 0)
		{
			a[j]=i;
			j++;
			n = n/i;
		}
	}
	if (n > 2)
	{
	    a[j]=n;
		
	    
	}
    
    int ab=0;

             ab+=a[i]%10;

             //cout<<ab<<endl;
            }}
            
            else
            { ab+=a[i];
            // cout<<ab<<endl;
                
            }
    }
    
    int check=0;

    {
        check+=sum%10;

}
